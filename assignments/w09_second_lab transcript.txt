"""
Generate a random sentence.

Sentence format:
    <determiner> <noun> <verb> <prepositional phrase>.
"""

from __future__ import annotations

import random
from dataclasses import dataclass
from typing import Sequence


@dataclass(frozen=True)
class WordBank:
    determiners: Sequence[str]
    nouns: Sequence[str]
    verbs: Sequence[str]
    prepositional_phrases: Sequence[str]


DEFAULT_WORDS = WordBank(
    determiners=("Some", "Many", "A few", "A couple", "Sixty nine"),
    nouns=("cats", "dogs", "turtles", "kids", "pencils"),
    verbs=("thought", "died", "ate", "balled up", "will get rizzed up", "nae-naed", "skrrted"),
    prepositional_phrases=("on the game", "during his class", "in the fridge", "down the well", "on the court"),
)


def choose(rng: random.Random, options: Sequence[str]) -> str:
    """Return one random element from options."""
    if not options:
        raise ValueError("options must not be empty")
    return rng.choice(list(options))


def make_sentence(words: WordBank = DEFAULT_WORDS, rng: random.Random | None = None) -> str:
    """Create and return a randomized sentence."""
    rng = rng or random.Random()

    determiner = choose(rng, words.determiners)
    noun = choose(rng, words.nouns)
    verb = choose(rng, words.verbs)
    prep = choose(rng, words.prepositional_phrases)

    return f"{determiner} {noun} {verb} {prep}."


def main() -> None:
    print(make_sentence())


if __name__ == "__main__":
    main()

import random
from sentences import make_sentence

def test_make_sentence_is_deterministic_with_seed():
    rng = random.Random(123)
    s1 = make_sentence(rng=rng)

    rng = random.Random(123)
    s2 = make_sentence(rng=rng)

    assert s1 == s2
